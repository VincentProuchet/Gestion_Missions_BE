public class MissionDTO {

	private Integer id;
	private Date dateDebut;
	private Date dateFin;
    private Ville villeDepart;
    private Ville villeArrivee;
    private BigDecimal montantPrime;
    private Transport transport;
    private Statut statut; 
    private NatureDTO nature;
    private CollaborateurDTO collaborateur;
    private List<LigneFraisDTO> lignesFrais;

}

// Mission JSON :
{
	"id":00,
    "prime": montantPrime,
    "statut": statut,
    "transport": transport,
    "dateDebut": dateDebut,
    "dateFin": dateDebut,
    "villeDepart": villeDepart,
    "villeArrivee": villeArrivee
    "nature": nature,
    "collaborateur": collaborateur
    "lignesFrais": [...lignesFrais],
}


public class NatureDTO {
	
	private Integer id;
    private String libelle;
    private boolean hasPrime;
    private boolean hasFacture;
    private BigDecimal tjm;
    private float primePercent;
    private Date dateDebutValidite;
    private Date dateFinValidite;

}

// Nature JSON :
{
	"id":00,
    "libelle": libelle,
    "dateCreation": dateDebutValidite 
    "dateFin": dateFinValidite
    "prime": hasPrime,
    "facture": hasFacture,
    "tauxJournalier": tjm,
    "pourcentagePrime": hasPrime ? primePercent : undefined
}

{

    "prenom": prenom,

// Ligne de frais JSON :
